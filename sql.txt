  create table Musican
                      (MID number(6) not NULL,S_title varchar2(150),Name varchar2(50),City varchar2(100),Street varchar2(100),Phone number(11),
                       constraint MID_PK primary key (MID));
   
   
  create table Song 
                  (S_title varchar2(200) not NULL,Author varchar2(150),MID number(6),
                   constraint S_title_pk primary key(S_title));
                   
 create table Album 
                   (AlbumID number(6) not NULL,MID number(6),CopyRighDate DATE,Title varchar2(150),
                    constraint AlbumID_pk primary key (AlbumID));
                    
                    
  create table Instrument
                        (instrument_name varchar2(150),Musical_key varchar2(100),MID number(6),
                        constraint instrument_name_pk primary key (instrument_name));



 create table Musician_Instrument
                                (MID number(6),instrument_name  varchar2(150),
                                 constraint MI_PK primary key (MID, instrument_name));

create or replace view v_song_musical as
select m.Name as Musician_Name, s.S_title as Song_Title, s.Author as Song_Author
from Musican m
inner join Song s
on m.MID = s.MID;


create or replace view v_left_song_musical as
select m.Name as Musician_Name, s.S_title as Song_Title, s.Author as Song_Author
from Musican m
left join Song s
on m.MID = s.MID;


create or replace view v_right_song_musical as
select m.Name as Musician_Name, s.S_title as Song_Title, s.Author as Song_Author
from Musican m
right join Song s
on m.MID = s.MID;

create or replace view v_full_song_musical as
select m.Name as Musician_Name, s.S_title as Song_Title, s.Author as Song_Author
from Musican m
full outer join Song s
on m.MID = s.MID;

create or replace view v_cross_song_musical as
select m.Name as Musician_Name, i.instrument_name as Instrument_Name
from Musican m
cross join Instrument i;
